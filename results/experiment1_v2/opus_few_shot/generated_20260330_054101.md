## SQL — Queries

```sql
-- Query CF_QryEquipos: Para desplegables de equipos
SELECT e.CodigoEmpresa, e.CF_CodigoEquipo, e.CF_Nombre
FROM CF_Equipos e
WHERE e.CodigoEmpresa = @CodigoEmpresa
ORDER BY e.CF_Nombre

-- Query CF_QryResultados: Todos los resultados
SELECT r.CodigoEmpresa, r.CF_Jornada, r.CF_EquipoLocal, el.CF_Nombre AS NombreLocal,
       r.CF_GolesLocal, r.CF_EquipoVisitante, ev.CF_Nombre AS NombreVisitante, 
       r.CF_GolesVisitante, r.CF_Fecha
FROM CF_Resultados r
LEFT JOIN CF_Equipos el ON r.CodigoEmpresa = el.CodigoEmpresa AND r.CF_EquipoLocal = el.CF_CodigoEquipo
LEFT JOIN CF_Equipos ev ON r.CodigoEmpresa = ev.CodigoEmpresa AND r.CF_EquipoVisitante = ev.CF_CodigoEquipo
WHERE r.CodigoEmpresa = @CodigoEmpresa
ORDER BY r.CF_Jornada, r.CF_EquipoLocal

-- Query CF_QryClasificacion: Clasificación con nombres
SELECT c.CodigoEmpresa, c.CF_Posicion, c.CF_Equipo, e.CF_Nombre,
       c.CF_PartidosJugados, c.CF_PartidosGanados, c.CF_PartidosEmpatados,
       c.CF_PartidosPerdidos, c.CF_GolesFavor, c.CF_GolesContra,
       c.CF_Puntos, c.CF_Positivos, c.CF_DiferenciaGoles
FROM CF_Clasificacion c
LEFT JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo
WHERE c.CodigoEmpresa = @CodigoEmpresa
ORDER BY c.CF_Posicion

-- Query CF_QryEstadisticasEquipo: Para cálculo de clasificación
SELECT 
    e.CF_CodigoEquipo,
    e.CF_Nombre,
    -- Partidos como local
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo) +
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo) AS PJ,
    -- Partidos ganados
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante) +
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal) AS PG,
    -- Partidos empatados
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal = r.CF_GolesVisitante) +
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante = r.CF_GolesLocal) AS PE,
    -- Partidos perdidos
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante) +
    (SELECT COUNT(*) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante < r.CF_GolesLocal) AS PP,
    -- Goles a favor
    ISNULL((SELECT SUM(r.CF_GolesLocal) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo), 0) +
    ISNULL((SELECT SUM(r.CF_GolesVisitante) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo), 0) AS GF,
    -- Goles en contra
    ISNULL((SELECT SUM(r.CF_GolesVisitante) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoLocal = e.CF_CodigoEquipo), 0) +
    ISNULL((SELECT SUM(r.CF_GolesLocal) FROM CF_Resultados r 
     WHERE r.CodigoEmpresa = e.CodigoEmpresa AND r.CF_EquipoVisitante = e.CF_CodigoEquipo), 0) AS GC
FROM CF_Equipos e
WHERE e.CodigoEmpresa = @CodigoEmpresa
```

    Dim lJuegaEuropa As Campo(CF_Equipos, CF_JuegaEuropa)

    Ifn lJuegaEuropa = 0 Then

        FrmControl("CF_Equipos", FIELDVALUE, "CF_Competicion", "1")
    Else
        FrmControl("CF_Equipos", CONTROLENABLED, "CF_Competicion", "-1")

            ' Ya tiene valor válido

            lCompeticion = 2  ' Europa League por defecto
            FrmControl("CF_Equipos", FIELDVALUE, "CF_Competicion", "2")
        Endif
    Endif

## 4GL — Calculation Scripts

### CF_CalculaClasificacion Script

```4gl
' === Cálculo de Clasificación ===
Dim lSQL As String
Dim lEmpresa As Integer
Dim lPosicion As Integer

' Variables para estadísticas
Dim lCodigoEquipo As Integer
Dim lNombre As String
Dim lPJ As Integer
Dim lPG As Integer
Dim lPE As Integer
Dim lPP As Integer
Dim lGF As Integer
Dim lGC As Integer
Dim lPuntos As Integer
Dim lPositivos As Integer
Dim lDiferenciaGoles As Integer

' Variables para cálculo de positivos
Dim lPGLocal As Integer
Dim lPELocal As Integer
Dim lPPLocal As Integer
Dim lPGVisit As Integer
Dim lPEVisit As Integer
Dim lPPVisit As Integer

' Registro de trabajo
Dim rEquipo As Registro
Dim rClasif As Registro

lEmpresa = Apli.ApliCodigoEmpresa

' Limpia clasificación anterior
lSQL = "DELETE FROM CF_Clasificacion WHERE CodigoEmpresa = " & lEmpresa
EjecutaBorra("exec:=" & lSQL)

' Procesa cada equipo
AbreConsulta(rEquipo, "CF_Equipos", "CodigoEmpresa = " & lEmpresa, "CF_CodigoEquipo")
Primero(rEquipo)

Whilen Not Eof(rEquipo) Then
    lCodigoEquipo = rEquipo.CF_CodigoEquipo
    lNombre = rEquipo.CF_Nombre
    
    ' Calcula partidos como local
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoLocal = " & lCodigoEquipo & " AND CF_GolesLocal > CF_GolesVisitante"
    lPGLocal = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoLocal = " & lCodigoEquipo & " AND CF_GolesLocal = CF_GolesVisitante"
    lPELocal = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoLocal = " & lCodigoEquipo & " AND CF_GolesLocal < CF_GolesVisitante"
    lPPLocal = Val(EjecutaSQL("exec:=" & lSQL))
    
    ' Calcula partidos como visitante
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoVisitante = " & lCodigoEquipo & " AND CF_GolesVisitante > CF_GolesLocal"
    lPGVisit = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoVisitante = " & lCodigoEquipo & " AND CF_GolesVisitante = CF_GolesLocal"
    lPEVisit = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT COUNT(*) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoVisitante = " & lCodigoEquipo & " AND CF_GolesVisitante < CF_GolesLocal"
    lPPVisit = Val(EjecutaSQL("exec:=" & lSQL))
    
    ' Totales
    lPG = lPGLocal + lPGVisit
    lPE = lPELocal + lPEVisit
    lPP = lPPLocal + lPPVisit
    lPJ = lPG + lPE + lPP
    
    ' Goles a favor
    lSQL = "SELECT ISNULL(SUM(CF_GolesLocal),0) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoLocal = " & lCodigoEquipo
    lGF = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT ISNULL(SUM(CF_GolesVisitante),0) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoVisitante = " & lCodigoEquipo
    lGF = lGF + Val(EjecutaSQL("exec:=" & lSQL))
    
    ' Goles en contra
    lSQL = "SELECT ISNULL(SUM(CF_GolesVisitante),0) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoLocal = " & lCodigoEquipo
    lGC = Val(EjecutaSQL("exec:=" & lSQL))
    
    lSQL = "SELECT ISNULL(SUM(CF_GolesLocal),0) FROM CF_Resultados WHERE CodigoEmpresa = " & lEmpresa & " AND CF_EquipoVisitante = " & lCodigoEquipo
    lGC = lGC + Val(EjecutaSQL("exec:=" & lSQL))
    
    ' Cálculos
    lPuntos = (lPG * 3) + lPE
    lPositivos = (lPGVisit * 3) + lPEVisit + (lPELocal * -1) + (lPPLocal * -3)
    lDiferenciaGoles = lGF - lGC
    
    ' Inserta registro temporal con posición 0
    lSQL = "INSERT INTO CF_Clasificacion (CodigoEmpresa, CF_Posicion, CF_Equipo, CF_PartidosJugados, " & _
           "CF_PartidosGanados, CF_PartidosEmpatados, CF_PartidosPerdidos, CF_GolesFavor, CF_GolesContra, " & _
           "CF_Puntos, CF_Positivos, CF_DiferenciaGoles) VALUES (" & _
           lEmpresa & ", 0, " & lCodigoEquipo & ", " & lPJ & ", " & lPG & ", " & lPE & ", " & lPP & ", " & _
           lGF & ", " & lGC & ", " & lPuntos & ", " & lPositivos & ", " & lDiferenciaGoles & ")"
    
    EjecutaInserta("exec:=" & lSQL)
    
    Siguiente(rEquipo)
Loop

CierraRegistro(rEquipo)

' Actualiza posiciones según criterios de ordenación
lSQL = "SELECT CF_Equipo FROM CF_Clasificacion c " & _
       "INNER JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo " & _
       "WHERE c.CodigoEmpresa = " & lEmpresa & " " & _
       "ORDER BY c.CF_Puntos DESC, c.CF_Positivos DESC, c.CF_DiferenciaGoles DESC, c.CF_PartidosJugados ASC, e.CF_Nombre ASC"

AbreConsulta(rClasif, "SQL", lSQL, "")
Primero(rClasif)
lPosicion = 1

Whilen Not Eof(rClasif) Then
    lCodigoEquipo = rClasif.CF_Equipo
    
    lSQL = "UPDATE CF_Clasificacion SET CF_Posicion = " & lPosicion & _
           " WHERE CodigoEmpresa = " & lEmpresa & " AND CF_Equipo = " & lCodigoEquipo
    EjecutaModifica("exec:=" & lSQL)
    
    lPosicion = lPosicion + 1
    Siguiente(rClasif)
Loop

CierraRegistro(rClasif)

' Muestra mensaje y ejecuta informe
MsgBox("Clasificación calculada correctamente", 0, "Información")
ListadoEjecuta("CF_InformeClasificacion", "", "", "S", "", "")
```
