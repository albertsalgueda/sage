### CF_Resultados Query
```sql
-- Name: CF_Resultados
SELECT r.CodigoEmpresa, r.CF_Jornada, r.CF_EquipoLocal, el.CF_Nombre as NombreLocal,
       r.CF_GolesLocal, r.CF_EquipoVisitante, ev.CF_Nombre as NombreVisitante,
       r.CF_GolesVisitante, r.CF_Fecha, r.CF_IdResultado
FROM CF_Resultados r
LEFT JOIN CF_Equipos el ON r.CodigoEmpresa = el.CodigoEmpresa AND r.CF_EquipoLocal = el.CF_CodigoEquipo
LEFT JOIN CF_Equipos ev ON r.CodigoEmpresa = ev.CodigoEmpresa AND r.CF_EquipoVisitante = ev.CF_CodigoEquipo
ORDER BY r.CF_Jornada, r.CF_Fecha
```

### CF_Clasificacion Query
```sql
-- Name: CF_Clasificacion
SELECT c.CodigoEmpresa, c.CF_Posicion, c.CF_Equipo, e.CF_Nombre as NombreEquipo,
       c.CF_PartidosJugados, c.CF_PartidosGanados, c.CF_PartidosEmpatados,
       c.CF_PartidosPerdidos, c.CF_GolesFavor, c.CF_GolesContra,
       c.CF_Puntos, c.CF_Positivos, c.CF_DiferenciaGoles, c.CF_IdClasificacion
FROM CF_Clasificacion c
LEFT JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo
ORDER BY c.CF_Posicion
```

## 5. 4GL — Calculation Scripts

### CF_CalcularClasificacion Script
```
Dim Ret As Numero
Dim Equipos As Registro
Dim Resultados As Registro
Dim Clasificacion As Registro
Dim Contador As Numero
Dim CF_Eq.CodigoEmpresa As Campo (Equipos, CodigoEmpresa) Numero
Dim CF_Eq.CF_CodigoEquipo As Campo (Equipos, CF_CodigoEquipo) Numero
Dim CF_Eq.CF_Nombre As Campo (Equipos, CF_Nombre) Cadena
Dim CF_Res.CodigoEmpresa As Campo (Resultados, CodigoEmpresa) Numero
Dim CF_Res.CF_EquipoLocal As Campo (Resultados, CF_EquipoLocal) Numero
Dim CF_Res.CF_GolesLocal As Campo (Resultados, CF_GolesLocal) Numero
Dim CF_Res.CF_EquipoVisitante As Campo (Resultados, CF_EquipoVisitante) Numero
Dim CF_Res.CF_GolesVisitante As Campo (Resultados, CF_GolesVisitante) Numero
Dim CF_Cla.CodigoEmpresa As Campo (Clasificacion, CodigoEmpresa) Numero
Dim CF_Cla.CF_Posicion As Campo (Clasificacion, CF_Posicion) Numero
Dim CF_Cla.CF_Equipo As Campo (Clasificacion, CF_Equipo) Numero
Dim CF_Cla.CF_PartidosJugados As Campo (Clasificacion, CF_PartidosJugados) Numero
Dim CF_Cla.CF_PartidosGanados As Campo (Clasificacion, CF_PartidosGanados) Numero
Dim CF_Cla.CF_PartidosEmpatados As Campo (Clasificacion, CF_PartidosEmpatados) Numero
Dim CF_Cla.CF_PartidosPerdidos As Campo (Clasificacion, CF_PartidosPerdidos) Numero
Dim CF_Cla.CF_GolesFavor As Campo (Clasificacion, CF_GolesFavor) Numero
Dim CF_Cla.CF_GolesContra As Campo (Clasificacion, CF_GolesContra) Numero
Dim CF_Cla.CF_Puntos As Campo (Clasificacion, CF_Puntos) Numero
Dim CF_Cla.CF_Positivos As Campo (Clasificacion, CF_Positivos) Numero
Dim CF_Cla.CF_DiferenciaGoles As Campo (Clasificacion, CF_DiferenciaGoles) Numero
Dim EquipoActual As Numero
Dim PJ As Numero
Dim PG As Numero
Dim PE As Numero
Dim PP As Numero
Dim GF As Numero
Dim GC As Numero
Dim Puntos As Numero
Dim Positivos As Numero
Dim DG As Numero
Dim cadSQL As Cadena

TransAbre

' Limpiar tabla clasificacion
Ret = EjecutaBorra("CF_Clasificacion")

Equipos = AbreConsulta("CF_Equipos")
Resultados = AbreConsulta("CF_Resultados")
Clasificacion = AbreConsulta("CF_Clasificacion")

Ret = Seleccion(Equipos)
Ifn Ret = -1 Then
    Ret = Primero(Equipos)
    Whilen Ret = -1 Then
        EquipoActual = CF_Eq.CF_CodigoEquipo
        
        ' Inicializar contadores
        PJ = 0
        PG = 0
        PE = 0
        PP = 0
        GF = 0
        GC = 0
        Puntos = 0
        Positivos = 0
        
        ' Calcular estadisticas como local
        Ret = Seleccion(Resultados, "CF_EquipoLocal = " & EquipoActual)
        Ifn Ret = -1 Then
            Ret = Primero(Resultados)
            Whilen Ret = -1 Then
                PJ = PJ + 1
                GF = GF + CF_Res.CF_GolesLocal
                GC = GC + CF_Res.CF_GolesVisitante
                
                Ifn CF_Res.CF_GolesLocal > CF_Res.CF_GolesVisitante Then
                    PG = PG + 1
                    Puntos = Puntos + 3
                Else
                    Ifn CF_Res.CF_GolesLocal = CF_Res.CF_GolesVisitante Then
                        PE = PE + 1
                        Puntos = Puntos + 1
                        Positivos = Positivos - 1
                    Else
                        PP = PP + 1
                        Positivos = Positivos - 3
                    Endif
                Endif
                
                Ret = Siguiente(Resultados)
            Loop
        Endif
        
        ' Calcular estadisticas como visitante
        Ret = Seleccion(Resultados, "CF_EquipoVisitante = " & EquipoActual)
        Ifn Ret = -1 Then
            Ret = Primero(Resultados)
            Whilen Ret = -1 Then
                PJ = PJ + 1
                GF = GF + CF_Res.CF_GolesVisitante
                GC = GC + CF_Res.CF_GolesLocal
                
                Ifn CF_Res.CF_GolesVisitante > CF_Res.CF_GolesLocal Then
                    PG = PG + 1
                    Puntos = Puntos + 3
                    Positivos = Positivos + 3
                Else
                    Ifn CF_Res.CF_GolesVisitante = CF_Res.CF_GolesLocal Then
                        PE = PE + 1
                        Puntos = Puntos + 1
                        Positivos = Positivos + 1
                    Else
                        PP = PP + 1
                    Endif
                Endif
                
                Ret = Siguiente(Resultados)
            Loop
        Endif
        
        DG = GF - GC
        
        ' Insertar en clasificacion
        Ret = Seleccion(Clasificacion)
        IniciaRegistro(Clasificacion)
        CF_Cla.CF_Equipo = EquipoActual
        CF_Cla.CF_PartidosJugados = PJ
        CF_Cla.CF_PartidosGanados = PG
        CF_Cla.CF_PartidosEmpatados = PE
        CF_Cla.CF_PartidosPerdidos = PP
        CF_Cla.CF_GolesFavor = GF
        CF_Cla.CF_GolesContra = GC
        CF_Cla.CF_Puntos = Puntos
        CF_Cla.CF_Positivos = Positivos
        CF_Cla.CF_DiferenciaGoles = DG
        CF_Cla.CF_Posicion = 1
        Nuevo(Clasificacion)
        
        Ret = Siguiente(Equipos)
    Loop
Endif

' Ordenar y asignar posiciones
cadSQL = "UPDATE CF_Clasificacion SET CF_Posicion = ("
cadSQL = cadSQL & "SELECT COUNT(*) + 1 FROM CF_Clasificacion c2 "
cadSQL = cadSQL & "WHERE c2.CodigoEmpresa = CF_Clasificacion.CodigoEmpresa "
cadSQL = cadSQL & "AND (c2.CF_Puntos > CF_Clasificacion.CF_Puntos "
cadSQL = cadSQL & "OR (c2.CF_Puntos = CF_Clasificacion.CF_Puntos "
cadSQL = cadSQL & "AND c2.CF_Positivos > CF_Clasificacion.CF_Positivos) "
cadSQL = cadSQL & "OR (c2.CF_Puntos = CF_Clasificacion.CF_Puntos "
cadSQL = cadSQL & "AND c2.CF_Positivos = CF_Clasificacion.CF_Positivos "
cadSQL = cadSQL & "AND c2.CF_DiferenciaGoles > CF_Clasificacion.CF_DiferenciaGoles) "
cadSQL = cadSQL & "OR (c2.CF_Puntos = CF_Clasificacion.CF_Puntos "
cadSQL = cadSQL & "AND c2.CF_Positivos = CF_Clasificacion.CF_Positivos "
cadSQL = cadSQL & "AND c2.CF_DiferenciaGoles = CF_Clasificacion.CF_DiferenciaGoles "
cadSQL = cadSQL & "AND c2.CF_PartidosJugados < CF_Clasificacion.CF_PartidosJugados)))"

Ret = EjecutaSQL("exec:=" & cadSQL)

CierraRegistro(Equipos)
CierraRegistro(Resultados)
CierraRegistro(Clasificacion)

TransAcepta

MsgBox("Clasificación calculada correctamente", 0, "Cálculo Completado")

' Ejecutar informe
Ret = ListadoEjecuta("CF_InformeClasificacion", "", "CF_Posicion", "*", "", "")

End
```
