+    ' Partidos Ganados
+    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 1 "

+    ' Partidos Empatados

+    sql = sql & "AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1 ELSE 0 END) as PE, "
+    
+    ' Partidos Perdidos
+    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN 1 "

+    
+    ' Goles a Favor
