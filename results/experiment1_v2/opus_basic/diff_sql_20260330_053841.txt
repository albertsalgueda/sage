+WITH TeamStats AS (
+    SELECT 
+        e.CodigoEmpresa,
+        e.CF_CodigoEquipo as Equipo,
+        e.CF_Nombre,
+        -- Partidos Jugados
+        COUNT(DISTINCT CASE 
+            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo 
+            THEN CONCAT(r.CF_Jornada, '-', r.CF_EquipoLocal, '-', r.CF_EquipoVisitante) 
+        END) as PJ,
+        -- Partidos Ganados
+        SUM(CASE 
+            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 1
+            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 1
+            ELSE 0
+        END) as PG,
+        -- Partidos Empatados
+        SUM(CASE 
+            WHEN (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) 
+                AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1
+            ELSE 0
+        END) as PE,
+        -- Partidos Perdidos
+        SUM(CASE 
+            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN 1

+            ELSE 0
+        END) as PP,
+        -- Goles a Favor
+        SUM(CASE
