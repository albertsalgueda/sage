## SQL — Queries

```sql
-- CF_Equipos_Listado: Equipos ordenados por código
SELECT CodigoEmpresa, CF_CodigoEquipo, CF_Nombre, CF_JuegaEuropa, CF_Competicion
FROM CF_Equipos
WHERE CodigoEmpresa = [CodigoEmpresa]
ORDER BY CF_CodigoEquipo

-- CF_Resultados_Listado: Resultados por jornada
SELECT r.CodigoEmpresa, r.CF_Jornada, r.CF_EquipoLocal, el.CF_Nombre AS NombreLocal,
       r.CF_GolesLocal, r.CF_EquipoVisitante, ev.CF_Nombre AS NombreVisitante,
       r.CF_GolesVisitante, r.CF_Fecha
FROM CF_Resultados r
LEFT JOIN CF_Equipos el ON r.CodigoEmpresa = el.CodigoEmpresa AND r.CF_EquipoLocal = el.CF_CodigoEquipo
LEFT JOIN CF_Equipos ev ON r.CodigoEmpresa = ev.CodigoEmpresa AND r.CF_EquipoVisitante = ev.CF_CodigoEquipo
WHERE r.CodigoEmpresa = [CodigoEmpresa]
ORDER BY r.CF_Jornada, r.CF_Fecha

-- CF_Clasificacion_Ordenada: Clasificación ordenada por criterios
SELECT c.CodigoEmpresa, c.CF_Posicion, c.CF_Equipo, e.CF_Nombre,
       c.CF_PartidosJugados, c.CF_PartidosGanados, c.CF_PartidosEmpatados,
       c.CF_PartidosPerdidos, c.CF_GolesFavor, c.CF_GolesContra,
       c.CF_Puntos, c.CF_Positivos, c.CF_DiferenciaGoles
FROM CF_Clasificacion c
LEFT JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo
WHERE c.CodigoEmpresa = [CodigoEmpresa]
ORDER BY c.CF_Puntos DESC, c.CF_Positivos DESC, c.CF_DiferenciaGoles DESC,
         c.CF_PartidosJugados ASC, e.CF_Nombre ASC

-- CF_Equipos_Desplegable: Para desplegables de equipos
SELECT CF_CodigoEquipo, CF_Nombre
FROM CF_Equipos
WHERE CodigoEmpresa = [CodigoEmpresa]
ORDER BY CF_Nombre
```

## 4GL — Calculation Scripts

### CF_CalcularClasificacion Calculation Script
```4gl
' === CALCULO DE CLASIFICACIÓN DE LIGA ===
Dim CodigoEmpresa As Entero
Dim SQL As Texto
Dim Equipo As Entero
Dim NombreEquipo As Texto
Dim PJ, PG, PE, PP, GF, GC, Puntos, Positivos, DG As Entero
Dim GolesLocal, GolesVisitante As Entero
Dim Posicion As Entero

CodigoEmpresa = Apli.ApliCodigoEmpresa

' Limpiar tabla de clasificación
SQL = "DELETE FROM CF_Clasificacion WHERE CodigoEmpresa = " & CodigoEmpresa
EjecutaSQL("exec:=" & SQL)

' Procesar cada equipo
SQL = "SELECT CF_CodigoEquipo, CF_Nombre FROM CF_Equipos WHERE CodigoEmpresa = " & CodigoEmpresa & " ORDER BY CF_CodigoEquipo"
AbreConsulta("Equipos", "exec:=" & SQL)

Whilen Primero("Equipos") <> 0 Then
    Equipo = Campo("Equipos", "CF_CodigoEquipo")
    NombreEquipo = Campo("Equipos", "CF_Nombre")
    
    ' Inicializar contadores
    PJ = 0: PG = 0: PE = 0: PP = 0
    GF = 0: GC = 0: Puntos = 0: Positivos = 0
    
    ' Calcular estadísticas como local
    SQL = "SELECT CF_GolesLocal, CF_GolesVisitante FROM CF_Resultados WHERE CodigoEmpresa = " & CodigoEmpresa & " AND CF_EquipoLocal = " & Equipo
    AbreConsulta("PartidosLocal", "exec:=" & SQL)
    
    Whilen Primero("PartidosLocal") <> 0 Then
        GolesLocal = Campo("PartidosLocal", "CF_GolesLocal")
        GolesVisitante = Campo("PartidosLocal", "CF_GolesVisitante")
        
        PJ = PJ + 1
        GF = GF + GolesLocal
        GC = GC + GolesVisitante
        
        Ifn GolesLocal > GolesVisitante Then
            ' Victoria como local
            PG = PG + 1
            Puntos = Puntos + 3
            ' Como local: 0 positivos por ganar
        Else
            Ifn GolesLocal = GolesVisitante Then
                ' Empate como local
                PE = PE + 1
                Puntos = Puntos + 1
                Positivos = Positivos - 1  ' -1 por empatar en casa
            Else
                ' Derrota como local
                PP = PP + 1
                Positivos = Positivos - 3  ' -3 por perder en casa
            Endif
        Endif
        
        Siguiente("PartidosLocal")
    Loop
    CierraRegistro("PartidosLocal")
    
    ' Calcular estadísticas como visitante
    SQL = "SELECT CF_GolesLocal, CF_GolesVisitante FROM CF_Resultados WHERE CodigoEmpresa = " & CodigoEmpresa & " AND CF_EquipoVisitante = " & Equipo
    AbreConsulta("PartidosVisitante", "exec:=" & SQL)
    
    Whilen Primero("PartidosVisitante") <> 0 Then
        GolesLocal = Campo("PartidosVisitante", "CF_GolesLocal")
        GolesVisitante = Campo("PartidosVisitante", "CF_GolesVisitante")
        
        PJ = PJ + 1
        GF = GF + GolesVisitante
        GC = GC + GolesLocal
        
        Ifn GolesVisitante > GolesLocal Then
            ' Victoria como visitante
            PG = PG + 1
            Puntos = Puntos + 3
            Positivos = Positivos + 3  ' +3 por ganar fuera
        Else
            Ifn GolesVisitante = GolesLocal Then
                ' Empate como visitante
                PE = PE + 1
                Puntos = Puntos + 1
                Positivos = Positivos + 1  ' +1 por empatar fuera
            Else
                ' Derrota como visitante
                PP = PP + 1
                ' 0 positivos por perder fuera
            Endif
        Endif
        
        Siguiente("PartidosVisitante")
    Loop
    CierraRegistro("PartidosVisitante")
    
    ' Calcular diferencia de goles
    DG = GF - GC
    
    ' Insertar registro temporal en clasificación
    SQL = "INSERT INTO CF_Clasificacion (CodigoEmpresa, CF_Posicion, CF_Equipo, CF_PartidosJugados, CF_PartidosGanados, CF_PartidosEmpatados, CF_PartidosPerdidos, CF_GolesFavor, CF_GolesContra, CF_Puntos, CF_Positivos, CF_DiferenciaGoles) VALUES (" & CodigoEmpresa & ", 0, " & Equipo & ", " & PJ & ", " & PG & ", " & PE & ", " & PP & ", " & GF & ", " & GC & ", " & Puntos & ", " & Positivos & ", " & DG & ")"
    EjecutaSQL("exec:=" & SQL)
    
    Siguiente("Equipos")
Loop
CierraRegistro("Equipos")

' Asignar posiciones según criterios de desempate
Posicion = 1
SQL = "SELECT CF_Equipo, CF_Puntos, CF_Positivos, CF_DiferenciaGoles, CF_PartidosJugados FROM CF_Clasificacion c LEFT JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo WHERE c.CodigoEmpresa = " & CodigoEmpresa & " ORDER BY CF_Puntos DESC, CF_Positivos DESC, CF_DiferenciaGoles DESC, CF_PartidosJugados ASC, e.CF_Nombre ASC"
AbreConsulta("ClasificacionOrdenada", "exec:=" & SQL)

Whilen Primero("ClasificacionOrdenada") <> 0 Then
    Equipo = Campo("ClasificacionOrdenada", "CF_Equipo")
    
    ' Actualizar posición
    SQL = "UPDATE CF_Clasificacion SET CF_Posicion = " & Posicion & " WHERE CodigoEmpresa = " & CodigoEmpresa & " AND CF_Equipo = " & Equipo
    EjecutaSQL("exec:=" & SQL)
    
    Posicion = Posicion + 1
    Siguiente("ClasificacionOrdenada")
Loop
CierraRegistro("ClasificacionOrdenada")

MsgBox "Clasificación calculada correctamente"

' Ejecutar informe
ListadoEjecuta("CF_InformeClasificacion","","","S","","")
```
