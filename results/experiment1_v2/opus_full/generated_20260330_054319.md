```sql
-- Query: CF_Equipos
SELECT 
    CodigoEmpresa,
    CF_CodigoEquipo,
    CF_Nombre,
    CF_JuegaEuropa,
    CF_Competicion,
    IdCF_Equipo
FROM CF_Equipos
WHERE CodigoEmpresa = @@CodigoEmpresa@@
ORDER BY CF_CodigoEquipo

-- Query: CF_EquiposL (for lookups)
SELECT 
    CF_CodigoEquipo,
    CF_Nombre
FROM CF_Equipos
WHERE CodigoEmpresa = @@CodigoEmpresa@@
ORDER BY CF_Nombre

-- Query: CF_Resultados
SELECT 
    R.CodigoEmpresa,
    R.CF_Jornada,
    R.CF_EquipoLocal,
    EL.CF_Nombre AS NombreLocal,
    R.CF_GolesLocal,
    R.CF_EquipoVisitante,
    EV.CF_Nombre AS NombreVisitante,
    R.CF_GolesVisitante,
    R.CF_Fecha,
    R.IdCF_Resultado
FROM CF_Resultados R
LEFT JOIN CF_Equipos EL ON R.CodigoEmpresa = EL.CodigoEmpresa AND R.CF_EquipoLocal = EL.CF_CodigoEquipo
LEFT JOIN CF_Equipos EV ON R.CodigoEmpresa = EV.CodigoEmpresa AND R.CF_EquipoVisitante = EV.CF_CodigoEquipo
WHERE R.CodigoEmpresa = @@CodigoEmpresa@@
ORDER BY R.CF_Jornada, R.CF_EquipoLocal

-- Query: CF_Clasificacion
SELECT 
    C.CodigoEmpresa,
    C.CF_Posicion,
    C.CF_Equipo,
    E.CF_Nombre,
    C.CF_PartidosJugados,
    C.CF_PartidosGanados,
    C.CF_PartidosEmpatados,
    C.CF_PartidosPerdidos,
    C.CF_GolesFavor,
    C.CF_GolesContra,
    C.CF_Puntos,
    C.CF_Positivos,
    C.CF_DiferenciaGoles,
    C.IdCF_Clasificacion
FROM CF_Clasificacion C
LEFT JOIN CF_Equipos E ON C.CodigoEmpresa = E.CodigoEmpresa AND C.CF_Equipo = E.CF_CodigoEquipo
WHERE C.CodigoEmpresa = @@CodigoEmpresa@@
ORDER BY C.CF_Posicion

-- Query: CF_ResultadosCalculo (for classification calculation)
SELECT 
    R.CodigoEmpresa,
    R.CF_Jornada,
    R.CF_EquipoLocal,
    R.CF_GolesLocal,
    R.CF_EquipoVisitante,
    R.CF_GolesVisitante
FROM CF_Resultados R
WHERE R.CodigoEmpresa = @@CodigoEmpresa@@
ORDER BY R.CF_Jornada
```

### CF_CalcularClasificacion Script

```4gl
' CF_CalcularClasificacion - Calculate league standings
Dim Ret As Numero
Dim NumDel As Numero
Dim Pos As Numero
Dim cadSQL As Cadena

' Query handles
Dim Equipos As Registro
Dim Resultados As Registro
Dim Clasificacion As Registro

' Equipment fields
Dim EQ.CodigoEmpresa As Campo (Equipos, CodigoEmpresa) Numero
Dim EQ.CF_CodigoEquipo As Campo (Equipos, CF_CodigoEquipo) Numero
Dim EQ.CF_Nombre As Campo (Equipos, CF_Nombre) Cadena

' Results fields
Dim RES.CF_EquipoLocal As Campo (Resultados, CF_EquipoLocal) Numero
Dim RES.CF_GolesLocal As Campo (Resultados, CF_GolesLocal) Numero
Dim RES.CF_EquipoVisitante As Campo (Resultados, CF_EquipoVisitante) Numero
Dim RES.CF_GolesVisitante As Campo (Resultados, CF_GolesVisitante) Numero

' Classification fields
Dim CLA.CodigoEmpresa As Campo (Clasificacion, CodigoEmpresa) Numero
Dim CLA.CF_Posicion As Campo (Clasificacion, CF_Posicion) Numero
Dim CLA.CF_Equipo As Campo (Clasificacion, CF_Equipo) Numero
Dim CLA.CF_PartidosJugados As Campo (Clasificacion, CF_PartidosJugados) Numero
Dim CLA.CF_PartidosGanados As Campo (Clasificacion, CF_PartidosGanados) Numero
Dim CLA.CF_PartidosEmpatados As Campo (Clasificacion, CF_PartidosEmpatados) Numero
Dim CLA.CF_PartidosPerdidos As Campo (Clasificacion, CF_PartidosPerdidos) Numero
Dim CLA.CF_GolesFavor As Campo (Clasificacion, CF_GolesFavor) Numero
Dim CLA.CF_GolesContra As Campo (Clasificacion, CF_GolesContra) Numero
Dim CLA.CF_Puntos As Campo (Clasificacion, CF_Puntos) Numero
Dim CLA.CF_Positivos As Campo (Clasificacion, CF_Positivos) Numero
Dim CLA.CF_DiferenciaGoles As Campo (Clasificacion, CF_DiferenciaGoles) Numero

' Statistics variables
Dim PJ As Numero
Dim PG As Numero
Dim PE As Numero
Dim PP As Numero
Dim GF As Numero
Dim GC As Numero
Dim Puntos As Numero
Dim Positivos As Numero
Dim DifGoles As Numero
Dim CodigoEquipoActual As Numero

' Delete existing classification
NumDel = EjecutaBorra("CF_Clasificacion")

' Process each team
Equipos = AbreConsulta("CF_Equipos")
Ret = Seleccion(Equipos)
Ifn Ret = -1 Then
    Ret = Primero(Equipos)
    Whilen Ret = -1 Then
        CodigoEquipoActual = EQ.CF_CodigoEquipo
        
        ' Initialize counters
        PJ = 0
        PG = 0
        PE = 0
        PP = 0
        GF = 0
        GC = 0
        Puntos = 0
        Positivos = 0
        
        ' Process matches as local team
        Resultados = AbreConsulta("CF_ResultadosCalculo")
        Ret = Seleccion(Resultados, "CF_EquipoLocal = " & Str(CodigoEquipoActual))
        Ifn Ret = -1 Then
            Ret = Primero(Resultados)
            Whilen Ret = -1 Then
                PJ = PJ + 1
                GF = GF + RES.CF_GolesLocal
                GC = GC + RES.CF_GolesVisitante
                
                Ifn RES.CF_GolesLocal > RES.CF_GolesVisitante Then
                    ' Win as local
                    PG = PG + 1
                    Puntos = Puntos + 3
                    ' No positives for local win
                Else
                    Ifn RES.CF_GolesLocal = RES.CF_GolesVisitante Then
                        ' Draw as local
                        PE = PE + 1
                        Puntos = Puntos + 1
                        Positivos = Positivos - 1
                    Else
                        ' Loss as local
                        PP = PP + 1
                        Positivos = Positivos - 3
                    Endif
                Endif
                
                Ret = Siguiente(Resultados)
            Loop
        Endif
        CierraRegistro(Resultados)
        
        ' Process matches as visitor team
        Resultados = AbreConsulta("CF_ResultadosCalculo")
        Ret = Seleccion(Resultados, "CF_EquipoVisitante = " & Str(CodigoEquipoActual))
        Ifn Ret = -1 Then
            Ret = Primero(Resultados)
            Whilen Ret = -1 Then
                PJ = PJ + 1
                GF = GF + RES.CF_GolesVisitante
                GC = GC + RES.CF_GolesLocal
                
                Ifn RES.CF_GolesVisitante > RES.CF_GolesLocal Then
                    ' Win as visitor
                    PG = PG + 1
                    Puntos = Puntos + 3
                    Positivos = Positivos + 3
                Else
                    Ifn RES.CF_GolesVisitante = RES.CF_GolesLocal Then
                        ' Draw as visitor
                        PE = PE + 1
                        Puntos = Puntos + 1
                        Positivos = Positivos + 1
                    Else
                        ' Loss as visitor
                        PP = PP + 1
                        ' No positives for visitor loss
                    Endif
                Endif
                
                Ret = Siguiente(Resultados)
            Loop
        Endif
        CierraRegistro(Resultados)
        
        ' Calculate goal difference
        DifGoles = GF - GC
        
        ' Insert into classification table (temporarily with position 0)
        Clasificacion = AbreConsulta("CF_Clasificacion")
        Ret = Seleccion(Clasificacion)
        IniciaRegistro(Clasificacion)
        CLA.CodigoEmpresa = EQ.CodigoEmpresa
        CLA.CF_Posicion = 0
        CLA.CF_Equipo = CodigoEquipoActual
        CLA.CF_PartidosJugados = PJ
        CLA.CF_PartidosGanados = PG
        CLA.CF_PartidosEmpatados = PE
        CLA.CF_PartidosPerdidos = PP
        CLA.CF_GolesFavor = GF
        CLA.CF_GolesContra = GC
        CLA.CF_Puntos = Puntos
        CLA.CF_Positivos = Positivos
        CLA.CF_DiferenciaGoles = DifGoles
        Nuevo(Clasificacion)
        CierraRegistro(Clasificacion)
        
        ' Next team
        Ret = Siguiente(Equipos)
    Loop
Endif
CierraRegistro(Equipos)

' Update positions based on sorting criteria
cadSQL = "UPDATE CF_Clasificacion SET CF_Posicion = T2.NewPos "
cadSQL = cadSQL & "FROM CF_Clasificacion T1 "
cadSQL = cadSQL & "INNER JOIN ( "
cadSQL = cadSQL & "  SELECT CF_Equipo, ROW_NUMBER() OVER ( "
cadSQL = cadSQL & "    ORDER BY CF_Puntos DESC, CF_Positivos DESC, CF_DiferenciaGoles DESC, "
cadSQL = cadSQL & "    CF_PartidosJugados ASC, CF_Equipo ASC "
cadSQL = cadSQL & "  ) AS NewPos "
cadSQL = cadSQL & "  FROM CF_Clasificacion "
cadSQL = cadSQL & "  WHERE CodigoEmpresa = " & Str(EQ.CodigoEmpresa)
cadSQL = cadSQL & ") T2 ON T1.CF_Equipo = T2.CF_Equipo "
cadSQL = cadSQL & "WHERE T1.CodigoEmpresa = " & Str(EQ.CodigoEmpresa)

Ret = EjecutaSQL("exec:=" & cadSQL)

' Execute classification report
Ret = ListadoEjecuta("CF_Clasificacion", "", "", "*", "", "")

End
```
