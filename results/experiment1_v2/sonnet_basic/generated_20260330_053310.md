```sql
-- Query para obtener clasificación ordenada
-- Nombre: CF_ClasificacionOrdenada
SELECT 
    c.CodigoEmpresa,
    ROW_NUMBER() OVER (
        ORDER BY c.CF_Puntos DESC, 
                 c.CF_Positivos DESC, 
                 c.CF_DiferenciaGoles DESC, 
                 c.CF_PartidosJugados ASC, 
                 e.CF_Nombre ASC
    ) as NuevaPosicion,
    c.CF_Equipo,
    e.CF_Nombre,
    c.CF_PartidosJugados,
    c.CF_PartidosGanados,
    c.CF_PartidosEmpatados,
    c.CF_PartidosPerdidos,
    c.CF_GolesFavor,
    c.CF_GolesContra,
    c.CF_Puntos,
    c.CF_Positivos,
    c.CF_DiferenciaGoles
FROM CF_Clasificacion c
INNER JOIN CF_Equipos e ON c.CodigoEmpresa = e.CodigoEmpresa AND c.CF_Equipo = e.CF_CodigoEquipo
WHERE c.CodigoEmpresa = ?CodigoEmpresa
```

## 4GL — Calculation Scripts

### Script de Cálculo de Clasificación (CF_CalculoClasificacion)

```4gl
Dim CodigoEmpresa As Entero
Dim SQLText As Texto
Dim RegEquipo As Registro
Dim RegClasificacion As Registro

Evento Inicio
    CodigoEmpresa = Apli.ApliCodigoEmpresa
    
    ' Limpiar tabla de clasificación
    SQLText = "DELETE FROM CF_Clasificacion WHERE CodigoEmpresa = " & CodigoEmpresa
    EjecutaSQL("exec:=" & SQLText)
    
    ' Procesar cada equipo
    Dim RegEquipo.CodigoEmpresa As Campo(CF_Equipos, CodigoEmpresa) Entero
    Dim RegEquipo.CF_CodigoEquipo As Campo(CF_Equipos, CF_CodigoEquipo) Entero
    
    RegEquipo.CodigoEmpresa = CodigoEmpresa
    AbreConsulta(RegEquipo)
    
    Whilen Primero(RegEquipo) Then
        Gosub CalcularEquipo
        Siguiente(RegEquipo)
    Loop
    
    CierraRegistro(RegEquipo)
    
    ' Actualizar posiciones según criterios de desempate
    Gosub ActualizarPosiciones
    
    ' Ejecutar informe
    ListadoEjecuta("CF_InformeClasificacion","","","S","","")
    
    MsgBox("Cálculo de clasificación completado")
FinEvento

Subrutina CalcularEquipo
    Dim CodigoEquipo As Entero
    Dim PJ, PG, PE, PP, GF, GC, Puntos, Positivos As Entero
    
    CodigoEquipo = RegEquipo.CF_CodigoEquipo
    PJ = 0
    PG = 0
    PE = 0
    PP = 0
    GF = 0
    GC = 0
    Puntos = 0
    Positivos = 0
    
    ' Calcular estadísticas como local
    SQLText = "SELECT COUNT(*) as Partidos, " &
              "SUM(CF_GolesLocal) as GolesFavor, " &
              "SUM(CF_GolesVisitante) as GolesContra, " &
              "SUM(CASE WHEN CF_GolesLocal > CF_GolesVisitante THEN 1 ELSE 0 END) as Ganados, " &
              "SUM(CASE WHEN CF_GolesLocal = CF_GolesVisitante THEN 1 ELSE 0 END) as Empatados, " &
              "SUM(CASE WHEN CF_GolesLocal < CF_GolesVisitante THEN 1 ELSE 0 END) as Perdidos " &
              "FROM CF_Resultados " &
              "WHERE CodigoEmpresa = " & CodigoEmpresa & " AND CF_EquipoLocal = " & CodigoEquipo
    
    EjecutaSQL("exec:=" & SQLText)
    
    Ifn Apli.SQLResultado.RecordCount > 0 Then
        PJ = PJ + Apli.SQLResultado("Partidos")
        GF = GF + Apli.SQLResultado("GolesFavor")
        GC = GC + Apli.SQLResultado("GolesContra")
        PG = PG + Apli.SQLResultado("Ganados")
        PE = PE + Apli.SQLResultado("Empatados")
        PP = PP + Apli.SQLResultado("Perdidos")
        
        ' Puntos como local: Ganados*3 + Empatados*1
        Puntos = Puntos + (Apli.SQLResultado("Ganados") * 3) + Apli.SQLResultado("Empatados")
        
        ' Positivos como local: Ganados*0 + Empatados*(-1) + Perdidos*(-3)
        Positivos = Positivos + (Apli.SQLResultado("Empatados") * -1) + (Apli.SQLResultado("Perdidos") * -3)
    Endif
    
    ' Calcular estadísticas como visitante
    SQLText = "SELECT COUNT(*) as Partidos, " &
              "SUM(CF_GolesVisitante) as GolesFavor, " &
              "SUM(CF_GolesLocal) as GolesContra, " &
              "SUM(CASE WHEN CF_GolesVisitante > CF_GolesLocal THEN 1 ELSE 0 END) as Ganados, " &
              "SUM(CASE WHEN CF_GolesVisitante = CF_GolesLocal THEN 1 ELSE 0 END) as Empatados, " &
              "SUM(CASE WHEN CF_GolesVisitante < CF_GolesLocal THEN 1 ELSE 0 END) as Perdidos " &
              "FROM CF_Resultados " &
              "WHERE CodigoEmpresa = " & CodigoEmpresa & " AND CF_EquipoVisitante = " & CodigoEquipo
    
    EjecutaSQL("exec:=" & SQLText)
    
    Ifn Apli.SQLResultado.RecordCount > 0 Then
        PJ = PJ + Apli.SQLResultado("Partidos")
        GF = GF + Apli.SQLResultado("GolesFavor")
        GC = GC + Apli.SQLResultado("GolesContra")
        PG = PG + Apli.SQLResultado("Ganados")
        PE = PE + Apli.SQLResultado("Empatados")
        PP = PP + Apli.SQLResultado("Perdidos")
        
        ' Puntos como visitante: Ganados*3 + Empatados*1
        Puntos = Puntos + (Apli.SQLResultado("Ganados") * 3) + Apli.SQLResultado("Empatados")
        
        ' Positivos como visitante: Ganados*3 + Empatados*1 + Perdidos*0
        Positivos = Positivos + (Apli.SQLResultado("Ganados") * 3) + Apli.SQLResultado("Empatados")
    Endif
    
    ' Insertar en tabla de clasificación
    Dim RegClasificacion.CodigoEmpresa As Campo(CF_Clasificacion, CodigoEmpresa) Entero
    Dim RegClasificacion.CF_Posicion As Campo(CF_Clasificacion, CF_Posicion) Entero
    Dim RegClasificacion.CF_Equipo As Campo(CF_Clasificacion, CF_Equipo) Entero
    Dim RegClasificacion.CF_PartidosJugados As Campo(CF_Clasificacion, CF_PartidosJugados) Entero
    Dim RegClasificacion.CF_PartidosGanados As Campo(CF_Clasificacion, CF_PartidosGanados) Entero
    Dim RegClasificacion.CF_PartidosEmpatados As Campo(CF_Clasificacion, CF_PartidosEmpatados) Entero
    Dim RegClasificacion.CF_PartidosPerdidos As Campo(CF_Clasificacion, CF_PartidosPerdidos) Entero
    Dim RegClasificacion.CF_GolesFavor As Campo(CF_Clasificacion, CF_GolesFavor) Entero
    Dim RegClasificacion.CF_GolesContra As Campo(CF_Clasificacion, CF_GolesContra) Entero
    Dim RegClasificacion.CF_Puntos As Campo(CF_Clasificacion, CF_Puntos) Entero
    Dim RegClasificacion.CF_Positivos As Campo(CF_Clasificacion, CF_Positivos) Entero
    Dim RegClasificacion.CF_DiferenciaGoles As Campo(CF_Clasificacion, CF_DiferenciaGoles) Entero
    
    IniciaRegistro(RegClasificacion)
    RegClasificacion.CodigoEmpresa = CodigoEmpresa
    RegClasificacion.CF_Posicion = 0  ' Se actualizará después
    RegClasificacion.CF_Equipo = CodigoEquipo
    RegClasificacion.CF_PartidosJugados = PJ
    RegClasificacion.CF_PartidosGanados = PG
    RegClasificacion.CF_PartidosEmpatados = PE
    RegClasificacion.CF_PartidosPerdidos = PP
    RegClasificacion.CF_GolesFavor = GF
    RegClasificacion.CF_GolesContra = GC
    RegClasificacion.CF_Puntos = Puntos
    RegClasificacion.CF_Positivos = Positivos
    RegClasificacion.CF_DiferenciaGoles = GF - GC
    
    Nuevo(RegClasificacion)
    Graba(RegClasificacion)
    CierraRegistro(RegClasificacion)
Retorna

Subrutina ActualizarPosiciones
    ' Actualizar posiciones usando la consulta ordenada
    SQLText = "UPDATE c SET CF_Posicion = t.NuevaPosicion " &
              "FROM CF_Clasificacion c " &
              "INNER JOIN (" &
              "    SELECT CodigoEmpresa, CF_Equipo, " &
              "           ROW_NUMBER() OVER (" &
              "               ORDER BY CF_Puntos DESC, CF_Positivos DESC, " &
              "                        CF_DiferenciaGoles DESC, CF_PartidosJugados ASC, " &
              "                        (SELECT CF_Nombre FROM CF_Equipos WHERE CodigoEmpresa = c.CodigoEmpresa AND CF_CodigoEquipo = c.CF_Equipo) ASC" &
              "           ) as NuevaPosicion " &
              "    FROM CF_Clasificacion c " &
              "    WHERE CodigoEmpresa = " & CodigoEmpresa &
              ") t ON c.CodigoEmpresa = t.CodigoEmpresa AND c.CF_Equipo = t.CF_Equipo " &
              "WHERE c.CodigoEmpresa = " & CodigoEmpresa
    
    EjecutaSQL("exec:=" & SQLText)
Retorna
```
