## SQL — Queries

```sql
-- Query for team lookups (used in Results screen)
SELECT CF_CodigoEquipo, CF_Nombre 
FROM CF_Equipos 
WHERE CodigoEmpresa = {CodigoEmpresa}
ORDER BY CF_Nombre

-- Query for ranking calculation with all tie-breaking criteria
WITH TeamStats AS (
    SELECT 
        e.CodigoEmpresa,
        e.CF_CodigoEquipo as Equipo,
        e.CF_Nombre,
        -- Partidos Jugados
        COUNT(DISTINCT CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo 
            THEN CONCAT(r.CF_Jornada, '-', r.CF_EquipoLocal, '-', r.CF_EquipoVisitante) 
        END) as PJ,
        -- Partidos Ganados
        SUM(CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 1
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 1
            ELSE 0
        END) as PG,
        -- Partidos Empatados
        SUM(CASE 
            WHEN (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) 
                AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1
            ELSE 0
        END) as PE,
        -- Partidos Perdidos
        SUM(CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN 1
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante < r.CF_GolesLocal THEN 1
            ELSE 0
        END) as PP,
        -- Goles a Favor
        SUM(CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo THEN r.CF_GolesLocal
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo THEN r.CF_GolesVisitante
            ELSE 0
        END) as GF,
        -- Goles en Contra
        SUM(CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo THEN r.CF_GolesVisitante
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo THEN r.CF_GolesLocal
            ELSE 0
        END) as GC,
        -- Puntos (3 per win, 1 per draw)
        SUM(CASE 
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 3
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 3
            WHEN (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) 
                AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1
            ELSE 0
        END) as Puntos,
        -- Positivos (special away/home calculation)
        SUM(CASE 
            -- Away wins: +3
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 3
            -- Away draws: +1
            WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante = r.CF_GolesLocal THEN 1
            -- Home draws: -1
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal = r.CF_GolesVisitante THEN -1
            -- Home losses: -3
            WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN -3
            ELSE 0
        END) as Positivos
    FROM CF_Equipos e
    LEFT JOIN CF_Resultados r ON e.CodigoEmpresa = r.CodigoEmpresa 
        AND (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo)
    WHERE e.CodigoEmpresa = {CodigoEmpresa}
    GROUP BY e.CodigoEmpresa, e.CF_CodigoEquipo, e.CF_Nombre
),
RankedTeams AS (
    SELECT 
        *,
        GF - GC as DiferenciaGoles,
        ROW_NUMBER() OVER (
            ORDER BY 
                Puntos DESC,
                Positivos DESC,
                (GF - GC) DESC,
                PJ ASC,
                CF_Nombre ASC
        ) as Posicion
    FROM TeamStats
)
SELECT * FROM RankedTeams
ORDER BY Posicion
```

```4gl
' CF_Equipos Screen Script

Sub Inicio()
    ' Set default values
    If Apli.ControlFormulari.Modo = "A" Then
        FrmControl("CF_Equipos", FIELDVALUE, "txtCF_JuegaEuropa", "0")
        FrmControl("CF_Equipos", FIELDVALUE, "txtCF_Competicion", "0")
        FrmControl("CF_Equipos", CONTROLENABLED, "txtCF_Competicion", "0")
    End If
End Sub

Sub AlCambiar()
    Dim campo As String
    campo = Apli.ControlFormulari.CampoConFoco
    
    If campo = "txtCF_JuegaEuropa" Then
        Dim juegaEuropa As String
        juegaEuropa = FrmControl("CF_Equipos", FIELDVALUE, "txtCF_JuegaEuropa")
        
        If juegaEuropa = "0" Then
            ' Disable competition field and clear it
            FrmControl("CF_Equipos", CONTROLENABLED, "txtCF_Competicion", "0")
            FrmControl("CF_Equipos", FIELDVALUE, "txtCF_Competicion", "0")
        Else
            ' Enable competition field
            FrmControl("CF_Equipos", CONTROLENABLED, "txtCF_Competicion", "-1")
            ' Set required
            FrmControl("CF_Equipos", REQUIREDVALUE, "txtCF_Competicion", "-1")
        End If
    End If
End Sub

Sub AntesInsertar()
    Dim juegaEuropa As String
    Dim competicion As String
    
    juegaEuropa = FrmControl("CF_Equipos", FIELDVALUE, "txtCF_JuegaEuropa")
    competicion = FrmControl("CF_Equipos", FIELDVALUE, "txtCF_Competicion")
    
    ' Validate business rules
    If juegaEuropa <> "0" Then
        If competicion = "0" Or competicion = "" Then
            MsgBox("Si el equipo juega en Europa, debe seleccionar Europa League o Champions League", 16, "Error")
            Apli.CancelarAccion = "-1"
        End If
    End If
End Sub

Sub AntesModificar()
    ' Same validation as insert
    Gosub AntesInsertar
End Sub
```

```4gl
' CF_Resultados Screen Script

Sub Inicio()
    ' Set default values
    If Apli.ControlFormulari.Modo = "A" Then
        FrmControl("CF_Resultados", FIELDVALUE, "txtCF_GolesLocal", "0")
        FrmControl("CF_Resultados", FIELDVALUE, "txtCF_GolesVisitante", "0")
        FrmControl("CF_Resultados", FIELDVALUE, "txtCF_Fecha", Format(Ahora(), "dd/mm/yyyy"))
    End If
End Sub

Sub AlCambiar()
    Dim campo As String
    campo = Apli.ControlFormulari.CampoConFoco
    
    ' Validate Jornada (1-38)
    If campo = "txtCF_Jornada" Then
        Dim jornada As Integer
        jornada = Val(FrmControl("CF_Resultados", FIELDVALUE, "txtCF_Jornada"))
        
        If jornada < 1 Or jornada > 38 Then
            MsgBox("La jornada debe estar entre 1 y 38", 16, "Error")
            FrmControl("CF_Resultados", FIELDVALUE, "txtCF_Jornada", "")
        End If
    End If
    
    ' Update team names when selecting teams
    If campo = "txtCF_EquipoLocal" Then
        Dim equipoLocal As String
        Dim nombreLocal As String
        equipoLocal = FrmControl("CF_Resultados", FIELDVALUE, "txtCF_EquipoLocal")
        
        If equipoLocal <> "" And equipoLocal <> "0" Then
            Dim sql As String
            sql = "SELECT CF_Nombre FROM CF_Equipos WHERE CodigoEmpresa = " & Apli.ApliCodigoEmpresa
            sql = sql & " AND CF_CodigoEquipo = " & equipoLocal
            
            Dim rs As Registro
            AbreConsulta(rs)
            rs.SQL = sql
            Seleccion(rs)
            If Not rs.EOF Then
                nombreLocal = rs("CF_Nombre")
                FrmControl("CF_Resultados", FIELDVALUE, "txtNombreLocal", nombreLocal)
            End If
            CierraRegistro(rs)
        End If
    End If
    
    If campo = "txtCF_EquipoVisitante" Then
        Dim equipoVisitante As String
        Dim nombreVisitante As String
        equipoVisitante = FrmControl("CF_Resultados", FIELDVALUE, "txtCF_EquipoVisitante")
        
        If equipoVisitante <> "" And equipoVisitante <> "0" Then
            Dim sql2 As String
            sql2 = "SELECT CF_Nombre FROM CF_Equipos WHERE CodigoEmpresa = " & Apli.ApliCodigoEmpresa
            sql2 = sql2 & " AND CF_CodigoEquipo = " & equipoVisitante
            
            Dim rs2 As Registro
            AbreConsulta(rs2)
            rs2.SQL = sql2
            Seleccion(rs2)
            If Not rs2.EOF Then
                nombreVisitante = rs2("CF_Nombre")
                FrmControl("CF_Resultados", FIELDVALUE, "txtNombreVisitante", nombreVisitante)
            End If
            CierraRegistro(rs2)
        End If
    End If
End Sub

Sub AntesInsertar()
    ' Validate teams are different
    Dim equipoLocal As String
    Dim equipoVisitante As String
    
    equipoLocal = FrmControl("CF_Resultados", FIELDVALUE, "txtCF_EquipoLocal")
    equipoVisitante = FrmControl("CF_Resultados", FIELDVALUE, "txtCF_EquipoVisitante")
    
    If equipoLocal = equipoVisitante Then
        MsgBox("El equipo local y visitante no pueden ser el mismo", 16, "Error")
        Apli.CancelarAccion = "-1"
    End If
End Sub

End Sub
```

### CF_CalcularClasificacion Process Script

```4gl
' CF_CalcularClasificacion - Calculate League Classification

Sub Main()
    Dim sql As String
    Dim codigoEmpresa As Integer
    codigoEmpresa = Apli.ApliCodigoEmpresa
    
    ' Clear existing classification
    sql = "DELETE FROM CF_Clasificacion WHERE CodigoEmpresa = " & codigoEmpresa
    EjecutaBorra(sql)
    
    ' Build the complex query for classification
    sql = "WITH TeamStats AS ( "
    sql = sql & "SELECT e.CodigoEmpresa, e.CF_CodigoEquipo as Equipo, e.CF_Nombre, "
    
    ' Partidos Jugados
    sql = sql & "COUNT(DISTINCT CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo "
    sql = sql & "THEN CONCAT(r.CF_Jornada, '-', r.CF_EquipoLocal, '-', r.CF_EquipoVisitante) END) as PJ, "
    
    ' Partidos Ganados
    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 1 "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 1 ELSE 0 END) as PG, "
    
    ' Partidos Empatados
    sql = sql & "SUM(CASE WHEN (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) "
    sql = sql & "AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1 ELSE 0 END) as PE, "
    
    ' Partidos Perdidos
    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN 1 "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante < r.CF_GolesLocal THEN 1 ELSE 0 END) as PP, "
    
    ' Goles a Favor
    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo THEN r.CF_GolesLocal "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo THEN r.CF_GolesVisitante ELSE 0 END) as GF, "
    
    ' Goles en Contra
    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo THEN r.CF_GolesVisitante "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo THEN r.CF_GolesLocal ELSE 0 END) as GC, "
    
    ' Puntos
    sql = sql & "SUM(CASE WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal > r.CF_GolesVisitante THEN 3 "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 3 "
    sql = sql & "WHEN (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) "
    sql = sql & "AND r.CF_GolesLocal = r.CF_GolesVisitante THEN 1 ELSE 0 END) as Puntos, "
    
    ' Positivos
    sql = sql & "SUM(CASE WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante > r.CF_GolesLocal THEN 3 "
    sql = sql & "WHEN r.CF_EquipoVisitante = e.CF_CodigoEquipo AND r.CF_GolesVisitante = r.CF_GolesLocal THEN 1 "
    sql = sql & "WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal = r.CF_GolesVisitante THEN -1 "
    sql = sql & "WHEN r.CF_EquipoLocal = e.CF_CodigoEquipo AND r.CF_GolesLocal < r.CF_GolesVisitante THEN -3 ELSE 0 END) as Positivos "
    
    sql = sql & "FROM CF_Equipos e LEFT JOIN CF_Resultados r ON e.CodigoEmpresa = r.CodigoEmpresa "
    sql = sql & "AND (r.CF_EquipoLocal = e.CF_CodigoEquipo OR r.CF_EquipoVisitante = e.CF_CodigoEquipo) "
    sql = sql & "WHERE e.CodigoEmpresa = " & codigoEmpresa & " "
    sql = sql & "GROUP BY e.CodigoEmpresa, e.CF_CodigoEquipo, e.CF_Nombre) "
    
    ' Insert into classification table
    sql = sql & "INSERT INTO CF_Clasificacion "
    sql = sql & "(CodigoEmpresa, CF_Posicion, CF_Equipo, CF_PartidosJugados, CF_PartidosGanados, "
    sql = sql & "CF_PartidosEmpatados, CF_PartidosPerdidos, CF_GolesFavor, CF_GolesContra, "
    sql = sql & "CF_Puntos, CF_Positivos, CF_DiferenciaGoles) "
    sql = sql & "SELECT CodigoEmpresa, "
    sql = sql & "ROW_NUMBER() OVER (ORDER BY Puntos DESC, Positivos DESC, (GF - GC) DESC, PJ ASC, CF_Nombre ASC) as Posicion, "
    sql = sql & "Equipo, PJ, PG, PE, PP, GF, GC, Puntos, Positivos, (GF - GC) as DiferenciaGoles "
    sql = sql & "FROM TeamStats"
    
    EjecutaInserta(sql)
    
    MsgBox("Clasificación calculada correctamente", 64, "Proceso completado")
    
    ' Launch classification report
    ListadoEjecuta("CF_InformeClasificacion", "", "", "S", "", "")
End Sub
```
