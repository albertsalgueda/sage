```sql
-- Query para calcular clasificación ordenada
SELECT 
    ROW_NUMBER() OVER (
        ORDER BY CF_Puntos DESC, CF_Positivos DESC, CF_DiferenciaGoles DESC, 
        CF_PartidosJugados ASC, E.CF_Nombre ASC
    ) as NuevaPosicion,
    C.CF_Equipo,
    C.CF_PartidosJugados,
    C.CF_PartidosGanados,
    C.CF_PartidosEmpatados,
    C.CF_PartidosPerdidos,
    C.CF_GolesFavor,
    C.CF_GolesContra,
    C.CF_Puntos,
    C.CF_Positivos,
    C.CF_DiferenciaGoles,
    E.CF_Nombre
FROM CF_Clasificacion C
INNER JOIN CF_Equipos E ON C.CodigoEmpresa = E.CodigoEmpresa AND C.CF_Equipo = E.CF_CodigoEquipo
WHERE C.CodigoEmpresa = ?
ORDER BY CF_Puntos DESC, CF_Positivos DESC, CF_DiferenciaGoles DESC, 
         CF_PartidosJugados ASC, E.CF_Nombre ASC
```

    Retorno

'=== Evento AntesInsertar ===
AntesInsertar:
    'Validar competición si juega en Europa

        Ifn RegEquipos.CF_Competicion = 0 Then
            MsgBox("Debe seleccionar una competición europea", 16, "Error")
            Apli.CancelarAccion = "-1"
            Retorno
        Endif
    Endif
    Retorno

'=== Evento AntesModificar ===
AntesModificar:
    'Validar competición si juega en Europa
    Ifn RegEquipos.CF_JuegaEuropa <> 0 Then
        Ifn RegEquipos.CF_Competicion = 0 Then
            MsgBox("Debe seleccionar una competición europea", 16, "Error")

Dim RegResultados As Registro
Dim RegResultados.CodigoEmpresa As Campo(CF_Resultados, CodigoEmpresa) smallint

## 4GL — Calculation Scripts

### CF_CalculoClasificacion Script
```4gl
'===== CF_CalculoClasificacion - Cálculo de Clasificación =====

'=== Variables ===
Dim RegEquipos As Registro
Dim RegEquipos.CF_CodigoEquipo As Campo(CF_Equipos, CF_CodigoEquipo) smallint
Dim RegEquipos.CF_Nombre As Campo(CF_Equipos, CF_Nombre) varchar

Dim RegResultados As Registro
Dim RegResultados.CF_Jornada As Campo(CF_Resultados, CF_Jornada) smallint
Dim RegResultados.CF_EquipoLocal As Campo(CF_Resultados, CF_EquipoLocal) smallint
Dim RegResultados.CF_GolesLocal As Campo(CF_Resultados, CF_GolesLocal) smallint
Dim RegResultados.CF_EquipoVisitante As Campo(CF_Resultados, CF_EquipoVisitante) smallint
Dim RegResultados.CF_GolesVisitante As Campo(CF_Resultados, CF_GolesVisitante) smallint

Dim RegClasif As Registro
Dim RegClasif.CF_Posicion As Campo(CF_Clasificacion, CF_Posicion) smallint
Dim RegClasif.CF_Equipo As Campo(CF_Clasificacion, CF_Equipo) smallint
Dim RegClasif.CF_PartidosJugados As Campo(CF_Clasificacion, CF_PartidosJugados) smallint
Dim RegClasif.CF_PartidosGanados As Campo(CF_Clasificacion, CF_PartidosGanados) smallint
Dim RegClasif.CF_PartidosEmpatados As Campo(CF_Clasificacion, CF_PartidosEmpatados) smallint
Dim RegClasif.CF_PartidosPerdidos As Campo(CF_Clasificacion, CF_PartidosPerdidos) smallint
Dim RegClasif.CF_GolesFavor As Campo(CF_Clasificacion, CF_GolesFavor) smallint
Dim RegClasif.CF_GolesContra As Campo(CF_Clasificacion, CF_GolesContra) smallint
Dim RegClasif.CF_Puntos As Campo(CF_Clasificacion, CF_Puntos) smallint
Dim RegClasif.CF_Positivos As Campo(CF_Clasificacion, CF_Positivos) smallint
Dim RegClasif.CF_DiferenciaGoles As Campo(CF_Clasificacion, CF_DiferenciaGoles) smallint

Dim CodigoEmpresa As smallint
Dim EquipoActual As smallint
Dim PJ, PG, PE, PP, GF, GC, Puntos, Positivos As smallint
Dim sql As varchar

'=== Inicio del Proceso ===
Inicio:
    CodigoEmpresa = Apli.ApliCodigoEmpresa
    
    'Borrar clasificación anterior
    sql = "DELETE FROM CF_Clasificacion WHERE CodigoEmpresa = " & CodigoEmpresa
    EjecutaSQL("exec:=" & sql)
    
    'Procesar cada equipo
    AbreConsulta(RegEquipos, "CF_Equipos", "CodigoEmpresa=" & CodigoEmpresa, "CF_CodigoEquipo")
    Whilen Primero(RegEquipos) = 0 Then
        EquipoActual = RegEquipos.CF_CodigoEquipo
        Gosub CalcularEstadisticasEquipo
        Gosub InsertarClasificacion
        Siguiente(RegEquipos)
    Loop
    CierraRegistro(RegEquipos)
    
    'Actualizar posiciones basadas en criterios de ordenación
    Gosub ActualizarPosiciones
    
    'Ejecutar informe
    ListadoEjecuta("CF_InformeClasificacion","","","S","","")
    
    MsgBox("Cálculo de clasificación completado", 64, "Información")
    Retorno

'=== Subrutinas ===
CalcularEstadisticasEquipo:
    'Inicializar contadores
    PJ = 0 : PG = 0 : PE = 0 : PP = 0
    GF = 0 : GC = 0 : Puntos = 0 : Positivos = 0
    
    'Calcular como local
    AbreConsulta(RegResultados, "CF_Resultados", "CodigoEmpresa=" & CodigoEmpresa & " AND CF_EquipoLocal=" & EquipoActual)
    Whilen Primero(RegResultados) = 0 Then
        PJ = PJ + 1
        GF = GF + RegResultados.CF_GolesLocal
        GC = GC + RegResultados.CF_GolesVisitante
        
        Ifn RegResultados.CF_GolesLocal > RegResultados.CF_GolesVisitante Then
            'Ganado como local
            PG = PG + 1
            Puntos = Puntos + 3
            'Como local: +0 positivos
        Else Ifn RegResultados.CF_GolesLocal = RegResultados.CF_GolesVisitante Then
            'Empatado como local
            PE = PE + 1
            Puntos = Puntos + 1
            Positivos = Positivos - 1  'Como local: -1 positivos
        Else
            'Perdido como local
            PP = PP + 1
            Positivos = Positivos - 3  'Como local: -3 positivos
        Endif
        
        Siguiente(RegResultados)
    Loop
    CierraRegistro(RegResultados)
    
    'Calcular como visitante
    AbreConsulta(RegResultados, "CF_Resultados", "CodigoEmpresa=" & CodigoEmpresa & " AND CF_EquipoVisitante=" & EquipoActual)
    Whilen Primero(RegResultados) = 0 Then
        PJ = PJ + 1
        GF = GF + RegResultados.CF_GolesVisitante
        GC = GC + RegResultados.CF_GolesLocal
        
        Ifn RegResultados.CF_GolesVisitante > RegResultados.CF_GolesLocal Then
            'Ganado como visitante
            PG = PG + 1
            Puntos = Puntos + 3
            Positivos = Positivos + 3  'Como visitante: +3 positivos
        Else Ifn RegResultados.CF_GolesVisitante = RegResultados.CF_GolesLocal Then
            'Empatado como visitante
            PE = PE + 1
            Puntos = Puntos + 1
            Positivos = Positivos + 1  'Como visitante: +1 positivos
        Else
            'Perdido como visitante
            PP = PP + 1
            'Como visitante: +0 positivos
        Endif
        
        Siguiente(RegResultados)
    Loop
    CierraRegistro(RegResultados)
    Retorno

InsertarClasificacion:
    Nuevo(RegClasif, "CF_Clasificacion")
    IniciaRegistro(RegClasif)
    RegClasif.CodigoEmpresa = CodigoEmpresa
    RegClasif.CF_Posicion = 0  'Se actualizará después
    RegClasif.CF_Equipo = EquipoActual
    RegClasif.CF_PartidosJugados = PJ
    RegClasif.CF_PartidosGanados = PG
    RegClasif.CF_PartidosEmpatados = PE
    RegClasif.CF_PartidosPerdidos = PP
    RegClasif.CF_GolesFavor = GF
    RegClasif.CF_GolesContra = GC
    RegClasif.CF_Puntos = Puntos
    RegClasif.CF_Positivos = Positivos
    RegClasif.CF_DiferenciaGoles = GF - GC
    Graba(RegClasif)
    CierraRegistro(RegClasif)
    Retorno

ActualizarPosiciones:
    'Usar query con ROW_NUMBER para ordenar correctamente
    sql = "UPDATE C SET CF_Posicion = T.NuevaPosicion " &_
          "FROM CF_Clasificacion C INNER JOIN (" &_
          "SELECT CF_Equipo, ROW_NUMBER() OVER (" &_
          "ORDER BY CF_Puntos DESC, CF_Positivos DESC, CF_DiferenciaGoles DESC, " &_
          "CF_PartidosJugados ASC, (SELECT CF_Nombre FROM CF_Equipos E WHERE E.CodigoEmpresa = C.CodigoEmpresa AND E.CF_CodigoEquipo = C.CF_Equipo) ASC" &_
          ") as NuevaPosicion " &_
          "FROM CF_Clasificacion C WHERE CodigoEmpresa = " & CodigoEmpresa &_
          ") T ON C.CF_Equipo = T.CF_Equipo " &_
          "WHERE C.CodigoEmpresa = " & CodigoEmpresa
    EjecutaSQL("exec:=" & sql)
    Retorno
```
